210 bool SaveFile(const std::string& sourcePath, const std::string& destinationPath)
211 {
212 std::ifstream sourceFile(sourcePath, std::ios::binary);
213 if (!sourceFile)
214 {
215 std::cout << "Failed to open source file: " << sourcePath << std::endl;
216 return false;
217 }
218
219 std::ofstream destinationFile(destinationPath, std::ios::binary);
220 if (!destinationFile)
221 {
222 std::cout << "Failed to create destination file: " << destinationPath << std::endl;
223 return false;
224 }
225
226 destinationFile << sourceFile.rdbuf();
227
228 if (sourceFile.bad())
229 {
230 std::cout << "Failed to read from source file: " << sourcePath << std::endl;
231 return false;
232 }
233
234 if (destinationFile.bad())
235 {
236 std::cout << "Failed to write to destination file: " << destinationPath << std::endl;
237 return false;
238 }
239 std::cout << "File saved successfully: " << sourcePath << " -> " << destinationPath << std::endl;
240 return true;
241 }
