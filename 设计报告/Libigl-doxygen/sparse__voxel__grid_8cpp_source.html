8template <typename DerivedP0, typename Func, typename DerivedS, typename DerivedV, typename DerivedI>
9IGL_INLINE void igl::sparse_voxel_grid(const Eigen::MatrixBase<DerivedP0>& p0,
10 const Func& scalarFunc,
11 const double eps,
12 const int expected_number_of_cubes,
13 Eigen::PlainObjectBase<DerivedS>& CS,
14 Eigen::PlainObjectBase<DerivedV>& CV,
15 Eigen::PlainObjectBase<DerivedI>& CI)
16{
17 typedef typename DerivedV::Scalar ScalarV;
18 typedef typename DerivedS::Scalar ScalarS;
19 typedef typename DerivedI::Scalar ScalarI;
20 typedef Eigen::Matrix<ScalarV, 1, 3> VertexRowVector;
21 typedef Eigen::Matrix<ScalarI, 1, 8> IndexRowVector;

24 struct IndexRowVectorHash {
25 std::size_t operator()(const Eigen::RowVector3i& key) const {
26 std::size_t seed = 0;
27 std::hash<int> hasher;
28 for (int i = 0; i < 3; i++) {
29 seed ^= hasher(key[i]) + 0x9e3779b9 + (seed<<6) + (seed>>2); // Copied from boost::hash_combine
30 }
31 return seed;
32 }
33 };
34
35 auto sgn = [](ScalarS val) -> int {
36 return (ScalarS(0) < val) - (val < ScalarS(0));
37 };
38
39 ScalarV half_eps = 0.5 * eps;
40
41 std::vector<IndexRowVector> CI_vector;
42 std::vector<VertexRowVector> CV_vector;
43 std::vector<ScalarS> CS_vector;
44 CI_vector.reserve(expected_number_of_cubes);
45 CV_vector.reserve(8 * expected_number_of_cubes);
46 CS_vector.reserve(8 * expected_number_of_cubes);
47
48 // Track visisted neighbors
49 std::unordered_map<Eigen::RowVector3i, int, IndexRowVectorHash> visited;
50 visited.reserve(6 * expected_number_of_cubes);
51 visited.max_load_factor(0.5);
52
53 // BFS Queue
54 std::vector<Eigen::RowVector3i> queue;
55 queue.reserve(expected_number_of_cubes * 8);
56 queue.push_back(Eigen::RowVector3i(0, 0, 0));
57
58 while (queue.size() > 0)
59 {
60 Eigen::RowVector3i pi = queue.back();
61 queue.pop_back();
62 if(visited.count(pi)){ continue; }
63
64 VertexRowVector ctr = p0 + eps*pi.cast<ScalarV>(); // R^3 center of this cube
65
66 // X, Y, Z basis vectors, and array of neighbor offsets used to construct cubes
67 const Eigen::RowVector3i bx(1, 0, 0), by(0, 1, 0), bz(0, 0, -1);
68 const std::array<Eigen::RowVector3i, 26> neighbors = {
69 bx, -bx, by, -by, bz, -bz,
70 by-bz, -by+bz, // 1-2 4-7
71 bx+by, -bx-by, // 0-1 7-6
72 by+bz, -by-bz, // 0-3 6-5
73 by-bx, -by+bx, // 2-3 5-4
74 bx-bz, -bx+bz, // 1-5 3-7
75 bx+bz, -bx-bz, // 0-4 2-6
76 -bx+by+bz, bx-by-bz, // 3 5
77 bx+by+bz, -bx-by-bz, // 0 6
78 bx+by-bz, -bx-by+bz, //1 7
79 -bx+by-bz, bx-by+bz, // 2 4,
80 };
81
82 // Compute the position of the cube corners and the scalar values at those corners
83 //
84 // Cube corners are ordered y-x-z, so their xyz offsets are:
85 //
86 // +++
87 // ++-
88 // -+-
89 // -++
90 // +-+
91 // +--
92 // ---
93 // --+
94 std::array<VertexRowVector, 8> cubeCorners = {
95 ctr+half_eps*(bx+by+bz).cast<ScalarV>(), ctr+half_eps*(bx+by-bz).cast<ScalarV>(), ctr+half_eps*(-bx+by-bz).cast<ScalarV>(), ctr+half_eps*(-bx+by+bz).cast<ScalarV>(),
96 ctr+half_eps*(bx-by+bz).cast<ScalarV>(), ctr+half_eps*(bx-by-bz).cast<ScalarV>(), ctr+half_eps*(-bx-by-bz).cast<ScalarV>(), ctr+half_eps*(-bx-by+bz).cast<ScalarV>()
97 };
98 std::array<ScalarS, 8> cubeScalars;
99 for (int i = 0; i < 8; i++) { cubeScalars[i] = scalarFunc(cubeCorners[i]); }
100
101 // If this cube doesn't intersect the surface, disregard it
102 bool validCube = false;
103 int sign = sgn(cubeScalars[0]);
104 for (int i = 1; i < 8; i++) {
105 if (sign != sgn(cubeScalars[i])) {
106 validCube = true;
107 break;
108 }
109 }
110 if (!validCube) {
111 continue;
112 }
113
114 // Add the cube vertices and indices to the output arrays if they are not there already
115 IndexRowVector cube;
116 uint8_t vertexAlreadyAdded = 0; // This is a bimask. If a bit is 1, it has been visited already by the BFS
117 constexpr std::array<uint8_t, 26> zv = {
118 (1 << 0) | (1 << 1) | (1 << 4) | (1 << 5),
119 (1 << 2) | (1 << 3) | (1 << 6) | (1 << 7),
120 (1 << 0) | (1 << 1) | (1 << 2) | (1 << 3),
121 (1 << 4) | (1 << 5) | (1 << 6) | (1 << 7),
122 (1 << 0) | (1 << 3) | (1 << 4) | (1 << 7),
123 (1 << 1) | (1 << 2) | (1 << 5) | (1 << 6),
124 (1 << 1) | (1 << 2),
125 (1 << 4) | (1 << 7),
126 (1 << 0) | (1 << 1),
127 (1 << 6) | (1 << 7),
128 (1 << 0) | (1 << 3),
129 (1 << 5) | (1 << 6),
130 (1 << 2) | (1 << 3),
131 (1 << 4) | (1 << 5),
132 (1 << 1) | (1 << 5),
133 (1 << 3) | (1 << 7),
134 (1 << 0) | (1 << 4),
135 (1 << 2) | (1 << 6),
136 (1 << 3), (1 << 5), // diagonals
137 (1 << 0), (1 << 6),
138 (1 << 1), (1 << 7),
139 (1 << 2), (1 << 4),
140 };
141 constexpr std::array<std::array<int, 4>, 26> zvv {{
142 {{0, 1, 4, 5}}, {{3, 2, 7, 6}}, {{0, 1, 2, 3}},
143 {{4, 5, 6, 7}}, {{0, 3, 4, 7}}, {{1, 2, 5, 6}},
144 {{-1,-1,1,2}}, {{-1,-1,4,7}}, {{-1,-1,0,1}},{{-1,-1,7,6}},
145 {{-1,-1,0,3}}, {{-1,-1,5,6}}, {{-1,-1,2,3}}, {{-1,-1,5,4}},
146 {{-1,-1,1,5}}, {{-1,-1,3,7}}, {{-1,-1,0,4}}, {{-1,-1,2,6}},
147 {{-1,-1,-1,3}}, {{-1,-1,-1,5}}, {{-1,-1,-1,0}}, {{-1,-1,-1,6}},
148 {{-1,-1,-1,1}}, {{-1,-1,-1,7}}, {{-1,-1,-1,2}}, {{-1,-1,-1,4}} }};
149
150 for (int n = 0; n < 26; n++) { // For each neighbor, check the hash table to see if its been added before
151 Eigen::RowVector3i nkey = pi + neighbors[n];
152 auto nbr = visited.find(nkey);
153 if (nbr != visited.end()) { // We've already visited this neighbor, use references to its vertices instead of duplicating them
154 vertexAlreadyAdded |= zv[n];
155 for (int i = 0; i < 4; i++)
156 {
157 if (zvv[n][i]!=-1)
158 {
159 cube[zvv[n][i]] = CI_vector[nbr->second][zvv[n % 2 == 0 ? n + 1 : n - 1][i]];
160 }
161 }
162 } else {
163 queue.push_back(nkey); // Otherwise, we have not visited the neighbor, put it in the BFS queue
164 }
165 }
166
167 for (int i = 0; i < 8; i++) { // Add new, non-visited,2 vertices to the arrays
168 if (0 == ((1 << i) & vertexAlreadyAdded)) {
169 cube[i] = CS_vector.size();
170 CV_vector.push_back(cubeCorners[i]);
171 CS_vector.push_back(cubeScalars[i]);
172 }
173 }
174
175 visited[pi] = CI_vector.size();
176 CI_vector.push_back(cube);
177 }
178
179 CV.conservativeResize(CV_vector.size(), 3);
180 CS.conservativeResize(CS_vector.size(), 1);
181 CI.conservativeResize(CI_vector.size(), 8);
182 // If you pass in column-major matrices, this is going to be slooooowwwww
183 for (int i = 0; i < CV_vector.size(); i++) {
184 CV.row(i) = CV_vector[i];
185 }
186 for (int i = 0; i < CS_vector.size(); i++) {
187 CS(i) = CS_vector[i];
188 }
189 for (int i = 0; i < CI_vector.size(); i++) {
190 CI.row(i) = CI_vector[i];
191 }
192}
