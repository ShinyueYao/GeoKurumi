21// Utility IGL_INLINE functions
22template <typename Q_type>
23static IGL_INLINE Q_type _QuatD(double w, double h)
24{
25 using namespace std;
26 return (Q_type)(std::abs(w) < std::abs(h) ? std::abs(w) : std::abs(h)) - 4;
27}
28template <typename Q_type>
29static IGL_INLINE Q_type _QuatIX(double x, double w, double h)
30{
31 return (2.0f*(Q_type)x - (Q_type)w - 1.0f)/_QuatD<Q_type>(w, h);
32}
33template <typename Q_type>
34static IGL_INLINE Q_type _QuatIY(double y, double w, double h)
35{
36 return (-2.0f*(Q_type)y + (Q_type)h - 1.0f)/_QuatD<Q_type>(w, h);
37}

39// This is largely the trackball as implemented in AntTweakbar. Much of the
40// code is straight from its source in TwMgr.cpp
41// http://www.antisphere.com/Wiki/tools:anttweakbar
42template <typename Q_type>
43IGL_INLINE void igl::trackball(
44 const double w,
45 const double h,
46 const Q_type speed_factor,
47 const double down_mouse_x,
48 const double down_mouse_y,
49 const double mouse_x,
50 const double mouse_y,
51 Q_type * quat)
52{
53 assert(speed_factor > 0);
54
55 double original_x =
56 _QuatIX<Q_type>(speed_factor*(down_mouse_x-w/2)+w/2, w, h);
57 double original_y =
58 _QuatIY<Q_type>(speed_factor*(down_mouse_y-h/2)+h/2, w, h);
59
60 double x = _QuatIX<Q_type>(speed_factor*(mouse_x-w/2)+w/2, w, h);
61 double y = _QuatIY<Q_type>(speed_factor*(mouse_y-h/2)+h/2, w, h);
62
63 double z = 1;
64 double n0 = sqrt(original_x*original_x + original_y*original_y + z*z);
65 double n1 = sqrt(x*x + y*y + z*z);
66 if(n0>igl::DOUBLE_EPS && n1>igl::DOUBLE_EPS)
67 {
68 double v0[] = { original_x/n0, original_y/n0, z/n0 };
69 double v1[] = { x/n1, y/n1, z/n1 };
70 double axis[3];
71 cross(v0,v1,axis);
72 double sa = sqrt(dot(axis, axis));
73 double ca = dot(v0, v1);
74 double angle = atan2(sa, ca);
75 if( x*x+y*y>1.0 )
76 {
77 angle *= 1.0 + 0.2f*(sqrt(x*x+y*y)-1.0);
78 }
79 double qrot[4];
80 axis_angle_to_quat(axis,angle,qrot);
81 quat[0] = qrot[0];
82 quat[1] = qrot[1];
83 quat[2] = qrot[2];
84 quat[3] = qrot[3];
85 }
86}

89template <typename Q_type>
90IGL_INLINE void igl::trackball(
91 const double w,
92 const double h,
93 const Q_type speed_factor,
94 const Q_type * down_quat,
95 const double down_mouse_x,
96 const double down_mouse_y,
97 const double mouse_x,
98 const double mouse_y,
99 Q_type * quat)
100{
101 double qrot[4], qres[4], qorig[4];
102 igl::trackball<double>(
103 w,h,
104 speed_factor,
105 down_mouse_x,down_mouse_y,
106 mouse_x,mouse_y,
107 qrot);
108 double nqorig =
109 sqrt(down_quat[0]*down_quat[0]+
110 down_quat[1]*down_quat[1]+
111 down_quat[2]*down_quat[2]+
112 down_quat[3]*down_quat[3]);
113
114 if( fabs(nqorig)>igl::DOUBLE_EPS_SQ )
115 {
116 qorig[0] = down_quat[0]/nqorig;
117 qorig[1] = down_quat[1]/nqorig;
118 qorig[2] = down_quat[2]/nqorig;
119 qorig[3] = down_quat[3]/nqorig;
120 igl::quat_mult<double>(qrot,qorig,qres);
121 quat[0] = qres[0];
122 quat[1] = qres[1];
123 quat[2] = qres[2];
124 quat[3] = qres[3];
125 }
126 else
127 {
128 quat[0] = qrot[0];
129 quat[1] = qrot[1];
130 quat[2] = qrot[2];
131 quat[3] = qrot[3];
132 }
133}

135template <typename Scalardown_quat, typename Scalarquat>
136IGL_INLINE void igl::trackball(
137 const double w,
138 const double h,
139 const double speed_factor,
140 const Eigen::Quaternion<Scalardown_quat> & down_quat,
141 const double down_mouse_x,
142 const double down_mouse_y,
143 const double mouse_x,
144 const double mouse_y,
145 Eigen::Quaternion<Scalarquat> & quat)
146{
147 using namespace std;
148 return trackball(
149 w,
150 h,
151 (Scalarquat)speed_factor,
152 down_quat.coeffs().data(),
153 down_mouse_x,
154 down_mouse_y,
155 mouse_x,
156 mouse_y,
157 quat.coeffs().data());
158}
