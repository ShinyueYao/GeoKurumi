8#include "qslim_optimal_collapse_edge_callbacks.h"
9#include "quadric_binary_plus_operator.h"
10#include <Eigen/LU>
11
12IGL_INLINE void igl::qslim_optimal_collapse_edge_callbacks(
13 Eigen::MatrixXi & E,
14 std::vector<std::tuple<Eigen::MatrixXd,Eigen::RowVectorXd,double> > &
15 quadrics,
16 int & v1,
17 int & v2,
18 decimate_cost_and_placement_callback & cost_and_placement,
19 decimate_pre_collapse_callback & pre_collapse,
20 decimate_post_collapse_callback & post_collapse)
21{
22 typedef std::tuple<Eigen::MatrixXd,Eigen::RowVectorXd,double> Quadric;
23 cost_and_placement = [&quadrics,&v1,&v2](
24 const int e,
25 const Eigen::MatrixXd & V,
26 const Eigen::MatrixXi & /*F*/,
27 const Eigen::MatrixXi & E,
28 const Eigen::VectorXi & /*EMAP*/,
29 const Eigen::MatrixXi & /*EF*/,
30 const Eigen::MatrixXi & /*EI*/,
31 double & cost,
32 Eigen::RowVectorXd & p)
33 {
34 // Combined quadric
35 Quadric quadric_p;
36 quadric_p = quadrics[E(e,0)] + quadrics[E(e,1)];
37 // Quadric: p'Ap + 2b'p + c
38 // optimal point: Ap = -b, or rather because we have row vectors: pA=-b
39 const auto & A = std::get<0>(quadric_p);
40 const auto & b = std::get<1>(quadric_p);
41 const auto & c = std::get<2>(quadric_p);
42 p = -b*A.inverse();
43 cost = p.dot(p*A) + 2*p.dot(b) + c;
44 // Force infs and nans to infinity
45 if(std::isinf(cost) || cost!=cost)
46 {
47 cost = std::numeric_limits<double>::infinity();
48 // Prevent NaNs. Actually NaNs might be useful for debugging.
49 p.setConstant(0);
50 }
51 };
52 // Remember endpoints
53 pre_collapse = [&v1,&v2](
54 const Eigen::MatrixXd & ,/*V*/
55 const Eigen::MatrixXi & ,/*F*/
56 const Eigen::MatrixXi & E ,
57 const Eigen::VectorXi & ,/*EMAP*/
58 const Eigen::MatrixXi & ,/*EF*/
59 const Eigen::MatrixXi & ,/*EI*/
60 const igl::min_heap< std::tuple<double,int,int> > & ,/*Q*/
61 const Eigen::VectorXi & ,/*EQ*/
62 const Eigen::MatrixXd & ,/*C*/
63 const int e)->bool
64 {
65 v1 = E(e,0);
66 v2 = E(e,1);
67 return true;
68 };
69 // update quadric
70 post_collapse = [&v1,&v2,&quadrics](
71 const Eigen::MatrixXd & , /*V*/
72 const Eigen::MatrixXi & , /*F*/
73 const Eigen::MatrixXi & , /*E*/
74 const Eigen::VectorXi & ,/*EMAP*/
75 const Eigen::MatrixXi & , /*EF*/
76 const Eigen::MatrixXi & , /*EI*/
77 const igl::min_heap< std::tuple<double,int,int> > & ,/*Q*/
78 const Eigen::VectorXi & ,/*EQ*/
79 const Eigen::MatrixXd & , /*C*/
80 const int , /*e*/
81 const int , /*e1*/
82 const int , /*e2*/
83 const int , /*f1*/
84 const int , /*f2*/
85 const bool collapsed
86 )->void
87 {
88 if(collapsed)
89 {
90 quadrics[v1<v2?v1:v2] = quadrics[v1] + quadrics[v2];
91 }
92 };
93}
