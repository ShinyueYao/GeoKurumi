17template <typename DerivedS, typename DerivedGV, typename DerivedV, typename DerivedF>
18IGL_INLINE void igl::marching_cubes(
19 const Eigen::MatrixBase<DerivedS> &S,
20 const Eigen::MatrixBase<DerivedGV> &GV,
21 const unsigned nx,
22 const unsigned ny,
23 const unsigned nz,
24 const typename DerivedS::Scalar isovalue,
25 Eigen::PlainObjectBase<DerivedV> &V,
26 Eigen::PlainObjectBase<DerivedF> &F)
27{
28 typedef typename DerivedS::Scalar Scalar;
29 typedef unsigned Index;
30 // use same order as a2fVertexOffset
31 const unsigned ioffset[8] = {0,1,1+nx,nx,nx*ny,1+nx*ny,1+nx+nx*ny,nx+nx*ny};
32
33
34 std::unordered_map<int64_t,int> E2V;
35 V.resize(std::pow(nx*ny*nz,2./3.),3);
36 F.resize(std::pow(nx*ny*nz,2./3.),3);
37 Index n = 0;
38 Index m = 0;
39
40 const auto xyz2i = [&nx,&ny,&nz]
41 (const int & x, const int & y, const int & z)->unsigned
42 {
43 return x+nx*(y+ny*(z));
44 };
45 const auto cube =
46 [
47 &GV,&S,&V,&n,&F,&m,&isovalue,
48 &E2V,&xyz2i,&ioffset
49 ]
50 (const int x, const int y, const int z)
51 {
52 const unsigned i = xyz2i(x,y,z);
53
54 //Make a local copy of the values at the cube's corners
55 static Eigen::Matrix<Scalar,8,1> cS;
56 static Eigen::Matrix<Index,8,1> cI;
57 //Find which vertices are inside of the surface and which are outside
58 for(int c = 0; c < 8; c++)
59 {
60 const unsigned ic = i + ioffset[c];
61 cI(c) = ic;
62 cS(c) = S(ic);
63 }
64
65 march_cube(GV,cS,cI,isovalue,V,n,F,m,E2V);
66
67 };
68
69 // march over all cubes (loop order chosen to match memory)
70 //
71 // Should be possible to parallelize safely if threads are "well separated".
72 // Like red-black Gauss Seidel. Probably each thread need's their own E2V,V,F,
73 // and then merge at the end. Annoying part are the edges lying on the
74 // interface between chunks.
75 for(int z=0;z<nz-1;z++)
76 {
77 for(int y=0;y<ny-1;y++)
78 {
79 for(int x=0;x<nx-1;x++)
80 {
81 cube(x,y,z);
82 }
83 }
84 }
85 V.conservativeResize(n,3);
86 F.conservativeResize(m,3);
87}

89template <
90 typename DerivedS,
91 typename DerivedGV,
92 typename DerivedGI,
93 typename DerivedV,
94 typename DerivedF>
95IGL_INLINE void igl::marching_cubes(
96 const Eigen::MatrixBase<DerivedS> & S,
97 const Eigen::MatrixBase<DerivedGV> & GV,
98 const Eigen::MatrixBase<DerivedGI> & GI,
99 const typename DerivedS::Scalar isovalue,
100 Eigen::PlainObjectBase<DerivedV> &V,
101 Eigen::PlainObjectBase<DerivedF> &F)
102{
103 typedef Eigen::Index Index;
104 typedef typename DerivedV::Scalar Scalar;
105
106 std::unordered_map<int64_t,int> E2V;
107 V.resize(4*GV.rows(),3);
108 F.resize(4*GV.rows(),3);
109 Index n = 0;
110 Index m = 0;
111
112 // march over cubes
113 for(Index c = 0;c<GI.rows();c++)
114 {
115 //Make a local copy of the values at the cube's corners
116 static Eigen::Matrix<Scalar,8,1> cS;
117 static Eigen::Matrix<Index,8,1> cI;
118 for(int v = 0; v < 8; v++)
119 {
120 cI(v) = GI(c,v);
121 cS(v) = S(GI(c,v));
122 }
123 march_cube(GV,cS,cI,isovalue,V,n,F,m,E2V);
124 }
125 V.conservativeResize(n,3);
126 F.conservativeResize(m,3);
127}
