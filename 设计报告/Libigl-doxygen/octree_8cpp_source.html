33 auto get_octant = [](const RowVector3PType& location,
34 const RowVector3CentersType& center){
35 // We use a binary numbering of children. Treating the parent cell's
36 // center as the origin, we number the octants in the following manner:
37 // The first bit is 1 iff the octant's x coordinate is positive
38 // The second bit is 1 iff the octant's y coordinate is positive
39 // The third bit is 1 iff the octant's z coordinate is positive
40 //
41 // For example, the octant with negative x, positive y, positive z is:
42 // 110 binary = 6 decimal
43 IndexType index = 0;
44 if( location(0) >= center(0)){
45 index = index + 1;
46 }
47 if( location(1) >= center(1)){
48 index = index + 2;
49 }
50 if( location(2) >= center(2)){
51 index = index + 4;
52 }
53 return index;
54 };

62 const CentersType h,
63 const ChildrenType child_index){
64 RowVector3CentersType change_vector;
65 change_vector << -h,-h,-h;
66
67 //positive x chilren are 1,3,4,7
68 if(child_index % 2){
69 change_vector(0) = h;
70 }
71 //positive y children are 2,3,6,7
72 if(child_index == 2 || child_index == 3 ||
73 child_index == 6 || child_index == 7){
74 change_vector(1) = h;
75 }
76 //positive z children are 4,5,6,7
77 if(child_index > 3){
78 change_vector(2) = h;
79 }
80 RowVector3CentersType output = parent_center + change_vector;
81 return output;

121 }

130 }
131 }
132 };

134 {

136 for(IndexType i = 0;i<all.size();i++) all[i]=i;
137 point_indices.emplace_back(all);
138 }
139 children.emplace_back(neg_ones);

141 //Get the minimum AABB for the points

143 RowVector3PType frontrighttop = P.colwise().maxCoeff();
144 RowVector3CentersType aabb_center = (backleftbottom+frontrighttop)/PointScalar(2.0);

146 centers.emplace_back( aabb_center );
147
148 //Widths are the side length of the cube, (not half the side length):
149 widths.emplace_back( aabb_width );
150 m++;

153
