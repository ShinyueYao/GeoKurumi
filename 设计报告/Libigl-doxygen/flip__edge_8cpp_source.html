11template <
12 typename DerivedF,
13 typename DerivedE,
14 typename DeriveduE,
15 typename DerivedEMAP,
16 typename uE2EType>
17IGL_INLINE void igl::flip_edge(
18 Eigen::PlainObjectBase<DerivedF> & F,
19 Eigen::PlainObjectBase<DerivedE> & E,
20 Eigen::PlainObjectBase<DeriveduE> & uE,
21 Eigen::PlainObjectBase<DerivedEMAP> & EMAP,
22 std::vector<std::vector<uE2EType> > & uE2E,
23 const size_t uei)
24{
25 typedef typename DerivedF::Scalar Index;
26 const size_t num_faces = F.rows();
27 assert(F.cols() == 3);
28 // Edge to flip [v1,v2] --> [v3,v4]
29 // Before:
30 // F(f1,:) = [v1,v2,v4] // in some cyclic order
31 // F(f2,:) = [v1,v3,v2] // in some cyclic order
32 // After:
33 // F(f1,:) = [v1,v3,v4] // in *this* order
34 // F(f2,:) = [v2,v4,v3] // in *this* order
35 //
36 // v1 v1
37 // /|\ / \
38 // / | \ /f1 \
39 // v3 /f2|f1\ v4 => v3 /_____\ v4
40 // \ | / \ f2 /
41 // \ | / \ /
42 // \|/ \ /
43 // v2 v2
44 auto& half_edges = uE2E[uei];
45 if (half_edges.size() != 2) {
46 throw "Cannot flip non-manifold or boundary edge";
47 }
48
49 const size_t f1 = half_edges[0] % num_faces;
50 const size_t f2 = half_edges[1] % num_faces;
51 const size_t c1 = half_edges[0] / num_faces;
52 const size_t c2 = half_edges[1] / num_faces;
53 assert(c1 < 3);
54 assert(c2 < 3);
55
56 assert(f1 != f2);
57 const size_t v1 = F(f1, (c1+1)%3);
58 const size_t v2 = F(f1, (c1+2)%3);
59 const size_t v4 = F(f1, c1);
60 const size_t v3 = F(f2, c2);
61 assert(F(f2, (c2+2)%3) == v1);
62 assert(F(f2, (c2+1)%3) == v2);
63
64 const size_t e_12 = half_edges[0];
65 const size_t e_24 = f1 + ((c1 + 1) % 3) * num_faces;
66 const size_t e_41 = f1 + ((c1 + 2) % 3) * num_faces;
67 const size_t e_21 = half_edges[1];
68 const size_t e_13 = f2 + ((c2 + 1) % 3) * num_faces;
69 const size_t e_32 = f2 + ((c2 + 2) % 3) * num_faces;
70 assert(E(e_12, 0) == v1);
71 assert(E(e_12, 1) == v2);
72 assert(E(e_24, 0) == v2);
73 assert(E(e_24, 1) == v4);
74 assert(E(e_41, 0) == v4);
75 assert(E(e_41, 1) == v1);
76 assert(E(e_21, 0) == v2);
77 assert(E(e_21, 1) == v1);
78 assert(E(e_13, 0) == v1);
79 assert(E(e_13, 1) == v3);
80 assert(E(e_32, 0) == v3);
81 assert(E(e_32, 1) == v2);
82
83 const size_t ue_24 = EMAP(e_24);
84 const size_t ue_41 = EMAP(e_41);
85 const size_t ue_13 = EMAP(e_13);
86 const size_t ue_32 = EMAP(e_32);
87
88 F(f1, 0) = v1;
89 F(f1, 1) = v3;
90 F(f1, 2) = v4;
91 F(f2, 0) = v2;
92 F(f2, 1) = v4;
93 F(f2, 2) = v3;
94
95 uE(uei, 0) = v3;
96 uE(uei, 1) = v4;
97
98 const size_t new_e_34 = f1;
99 const size_t new_e_41 = f1 + num_faces;
100 const size_t new_e_13 = f1 + num_faces*2;
101 const size_t new_e_43 = f2;
102 const size_t new_e_32 = f2 + num_faces;
103 const size_t new_e_24 = f2 + num_faces*2;
104
105 E(new_e_34, 0) = v3;
106 E(new_e_34, 1) = v4;
107 E(new_e_41, 0) = v4;
108 E(new_e_41, 1) = v1;
109 E(new_e_13, 0) = v1;
110 E(new_e_13, 1) = v3;
111 E(new_e_43, 0) = v4;
112 E(new_e_43, 1) = v3;
113 E(new_e_32, 0) = v3;
114 E(new_e_32, 1) = v2;
115 E(new_e_24, 0) = v2;
116 E(new_e_24, 1) = v4;
117
118 EMAP(new_e_34) = uei;
119 EMAP(new_e_43) = uei;
120 EMAP(new_e_41) = ue_41;
121 EMAP(new_e_13) = ue_13;
122 EMAP(new_e_32) = ue_32;
123 EMAP(new_e_24) = ue_24;
124
125 auto replace = [](std::vector<Index>& array, Index old_v, Index new_v) {
126 std::replace(array.begin(), array.end(), old_v, new_v);
127 };
128 replace(uE2E[uei], e_12, new_e_34);
129 replace(uE2E[uei], e_21, new_e_43);
130 replace(uE2E[ue_13], e_13, new_e_13);
131 replace(uE2E[ue_32], e_32, new_e_32);
132 replace(uE2E[ue_24], e_24, new_e_24);
133 replace(uE2E[ue_41], e_41, new_e_41);
134
135#ifndef NDEBUG
136 auto sanity_check = [&](size_t ue) {
137 const auto& adj_faces = uE2E[ue];
138 if (adj_faces.size() == 2) {
139 const size_t first_f = adj_faces[0] % num_faces;
140 const size_t first_c = adj_faces[0] / num_faces;
141 const size_t second_f = adj_faces[1] % num_faces;
142 const size_t second_c = adj_faces[1] / num_faces;
143 const size_t vertex_0 = F(first_f, (first_c+1) % 3);
144 const size_t vertex_1 = F(first_f, (first_c+2) % 3);
145 assert(vertex_0 == F(second_f, (second_c+2) % 3));
146 assert(vertex_1 == F(second_f, (second_c+1) % 3));
147 }
148 };
149
150 sanity_check(uei);
151 sanity_check(ue_13);
152 sanity_check(ue_32);
153 sanity_check(ue_24);
154 sanity_check(ue_41);
155#endif
156}
