12template <typename DerivedM>
13IGL_INLINE const Eigen::WithFormat< DerivedM > igl::matlab_format(
14 const Eigen::DenseBase<DerivedM> & M,
15 const std::string name)
16{
17 using namespace std;
18 string prefix = "";
19 if(!name.empty())
20 {
21 prefix = name + " = ";
22 }
23
24 return M.format(Eigen::IOFormat(
25 Eigen::FullPrecision,
26 0,
27 " ",
28 "\n",
29 "",
30 "",
31 // This seems like a bit of a hack since I would expect the rows to align
32 // with out this extra spacing on the first line
33 prefix + "[\n ",
34 "\n];"));
35}

37template <typename DerivedM>
38IGL_INLINE std::string igl::matlab_format_index(
39 const Eigen::MatrixBase<DerivedM> & M,
40 const std::string name)
41{
42 // can't return WithFormat since that uses a pointer to matrix
43 return STR(igl::matlab_format((M.array()+1).eval(),name));
44}

46template <typename DerivedS>
47IGL_INLINE const std::string
48igl::matlab_format(
49 const Eigen::SparseMatrix<DerivedS> & S,
50 const std::string name)
51{
52 using namespace Eigen;
53 using namespace std;
54 Matrix<typename Eigen::SparseMatrix<DerivedS>::Scalar,Dynamic,1> I,J,V;
55 Matrix<DerivedS,Dynamic,Dynamic> SIJV;
56 find(S,I,J,V);
57 I.array() += 1;
58 J.array() += 1;
59 SIJV.resize(V.rows(),3);
60 SIJV << I,J,V;
61 string prefix = "";
62 string suffix = "";
63 if(!name.empty())
64 {
65 prefix = name + "IJV = ";
66 suffix = "\n"+name + " = sparse("+name+"IJV(:,1),"+name+"IJV(:,2),"+name+"IJV(:,3),"+std::to_string(S.rows())+","+std::to_string(S.cols())+" );";
67 }
68 return STR(""<<
69 SIJV.format(Eigen::IOFormat(
70 Eigen::FullPrecision,
71 0,
72 " ",
73 "\n",
74 "",
75 "",
76 // This seems like a bit of a hack since I would expect the rows to align
77 // with out this extra spacing on the first line
78 prefix + "[\n ",
79 "\n];"))<<suffix);
80}

82IGL_INLINE Eigen::IOFormat igl::matlab_format()
83{
84 // M = [1 2 3;4 5 6];
85 // M.format(matlab_format()) produces:
86 // [
87 // 1 2 3
88 // 4 5 6
89 // ];
90 return Eigen::IOFormat(
91 Eigen::FullPrecision,
92 0,
93 " ",
94 "\n",
95 "",
96 "",
97 // This seems like a bit of a hack since I would expect the rows to align
98 // with out this extra spacing on the first line
99 "[\n ",
100 "\n];");
101}

103IGL_INLINE const std::string igl::matlab_format(
104 const double v,
105 const std::string name)
106{
107 using namespace std;
108 string prefix = "";
109 if(!name.empty())
110 {
111 prefix = name + " = ";
112 }
113 return STR(prefix<<v<<";");
114}
