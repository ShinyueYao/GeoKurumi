16// Static helper method reads the first to elements in the given file
17// Inputs:
18// fp file pointer of .dmat file that was just opened
19// Outputs:
20// num_rows number of rows
21// num_cols number of columns
22// Returns
23// 0 success
24// 1 did not find header
25// 2 bad num_cols
26// 3 bad num_rows
27// 4 bad line ending
28static inline int readDMAT_read_header(FILE * fp, int & num_rows, int & num_cols)
29{
30 // first line contains number of rows and number of columns
31 int res = fscanf(fp,"%d %d",&num_cols,&num_rows);
32 if(res != 2)
33 {
34 return 1;
35 }
36 // check that number of columns and rows are sane
37 if(num_cols < 0)
38 {
39 fprintf(stderr,"IOError: readDMAT() number of columns %d < 0\n",num_cols);
40 return 2;
41 }
42 if(num_rows < 0)
43 {
44 fprintf(stderr,"IOError: readDMAT() number of rows %d < 0\n",num_rows);
45 return 3;
46 }
47 // finish reading header
48 char lf;
49
50 if(fread(&lf, sizeof(char), 1, fp)!=1 || !(lf == '\n' || lf == '\r'))
51 {
52 fprintf(stderr,"IOError: bad line ending in header\n");
53 return 4;
54 }
55
56 return 0;
57}

60template <typename DerivedW>
61IGL_INLINE bool igl::readDMAT(const std::string file_name,
62 Eigen::PlainObjectBase<DerivedW> & W)
63{
64 FILE * fp = fopen(file_name.c_str(),"rb");
65 if(fp == NULL)
66 {
67 fprintf(stderr,"IOError: readDMAT() could not open %s...\n",file_name.c_str());
68 return false;
69 }
70 int num_rows,num_cols;
71 int head_success = readDMAT_read_header(fp,num_rows,num_cols);
72 if(head_success != 0)
73 {
74 if(head_success == 1)
75 {
76 fprintf(stderr,
77 "IOError: readDMAT() first row should be [num cols] [num rows]...\n");
78 }
79 fclose(fp);
80 return false;
81 }
82
83 // Resize output to fit matrix, only if non-empty since this will trigger an
84 // error on fixed size matrices before reaching binary data.
85 bool empty = num_rows == 0 || num_cols == 0;
86 if(!empty)
87 {
88 W.resize(num_rows,num_cols);
89 }
90
91 // Loop over columns slowly
92 for(int j = 0;j < num_cols;j++)
93 {
94 // loop over rows (down columns) quickly
95 for(int i = 0;i < num_rows;i++)
96 {
97 double d;
98 if(fscanf(fp," %lg",&d) != 1)
99 {
100 fclose(fp);
101 fprintf(
102 stderr,
103 "IOError: readDMAT() bad format after reading %d entries\n",
104 j*num_rows + i);
105 return false;
106 }
107 W(i,j) = d;
108 }
109 }
110
111 // Try to read header for binary part
112 head_success = readDMAT_read_header(fp,num_rows,num_cols);
113 if(head_success == 0)
114 {
115 assert(W.size() == 0);
116 // Resize for output
117 W.resize(num_rows,num_cols);
118 std::unique_ptr<double[]> Wraw(new double[num_rows*num_cols]);
119 fread(Wraw.get(), sizeof(double), num_cols*num_rows, fp);
120 // Loop over columns slowly
121 for(int j = 0;j < num_cols;j++)
122 {
123 // loop over rows (down columns) quickly
124 for(int i = 0;i < num_rows;i++)
125 {
126 W(i,j) = Wraw[j*num_rows+i];
127 }
128 }
129 }else
130 {
131 // we skipped resizing before in case there was binary data
132 if(empty)
133 {
134 // This could trigger an error if using fixed size matrices.
135 W.resize(num_rows,num_cols);
136 }
137 }
138
139 fclose(fp);
140 return true;
141}

188 }
189 }
190

192 head_success = readDMAT_read_header(fp,num_rows,num_cols);
