13template <
14 typename Derivedwin,
15 typename Derivedmodel,
16 typename Derivedproj,
17 typename Derivedviewport,
18 typename Derivedscene>
19IGL_INLINE void igl::unproject(
20 const Eigen::MatrixBase<Derivedwin>& win,
21 const Eigen::MatrixBase<Derivedmodel>& model,
22 const Eigen::MatrixBase<Derivedproj>& proj,
23 const Eigen::MatrixBase<Derivedviewport>& viewport,
24 Eigen::PlainObjectBase<Derivedscene> & scene)
25{
26 if(win.cols() != 3)
27 {
28 assert(win.rows() == 3);
29 // needless transposes
30 Eigen::Matrix<typename Derivedscene::Scalar,1,3> sceneT;
31 unproject(win.transpose().eval(),model,proj,viewport,sceneT);
32 scene = sceneT.head(3);
33 return;
34 }
35 assert(win.cols() == 3);
36 const int n = win.rows();
37 scene.resize(n,3);
38 for(int i = 0;i<n;i++)
39 {
40 typedef typename Derivedscene::Scalar Scalar;
41 Eigen::Matrix<Scalar,4,4> Inverse =
42 (proj.template cast<Scalar>() * model.template cast<Scalar>()).inverse();
43
44 Eigen::Matrix<Scalar,4,1> tmp;
45 tmp << win.row(i).head(3).transpose(), 1;
46 tmp(0) = (tmp(0) - viewport(0, 0)) / viewport(2, 0);
47 tmp(1) = (tmp(1) - viewport(1, 0)) / viewport(3, 0);
48 tmp = tmp.array() * 2.0f - 1.0f;
49
50 Eigen::Matrix<Scalar,4,1> obj = Inverse * tmp;
51 obj /= obj(3);
52
53 scene.row(i).head(3) = obj.head(3);
54 }
55}

57template <typename Scalar>
58IGL_INLINE Eigen::Matrix<Scalar,3,1> igl::unproject(
59 const Eigen::Matrix<Scalar,3,1>& win,
60 const Eigen::Matrix<Scalar,4,4>& model,
61 const Eigen::Matrix<Scalar,4,4>& proj,
62 const Eigen::Matrix<Scalar,4,1>& viewport)
63{
64 Eigen::Matrix<Scalar,3,1> scene;
65 unproject(win,model,proj,viewport,scene);
66 return scene;
67}
