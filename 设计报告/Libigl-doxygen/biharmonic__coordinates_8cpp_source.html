33template <
34 typename DerivedV,
35 typename DerivedT,
36 typename SType,
37 typename DerivedW>
38IGL_INLINE bool igl::biharmonic_coordinates(
39 const Eigen::MatrixBase<DerivedV> & V,
40 const Eigen::MatrixBase<DerivedT> & T,
41 const std::vector<std::vector<SType> > & S,
42 const int k,
43 Eigen::PlainObjectBase<DerivedW> & W)
44{
45 using namespace Eigen;
46 using namespace std;
47
48 typedef typename DerivedV::Scalar Scalar;
49 typedef typename DerivedT::Scalar Integer;
50
51 // This is not the most efficient way to build A, but follows "Linear
52 // Subspace Design for Real-Time Shape Deformation" [Wang et al. 2015].
53 SparseMatrix<Scalar> A;
54 {
55 DiagonalMatrix<Scalar, Dynamic> Minv;
56 SparseMatrix<Scalar> L, K;
57 Array<bool,Dynamic,Dynamic> C;
58 {
59 Array<bool,Dynamic,1> I;
60 on_boundary(T,I,C);
61 }
62#ifdef false
63 // Version described in paper is "wrong"
64 // http://www.cs.toronto.edu/~jacobson/images/error-in-linear-subspace-design-for-real-time-shape-deformation-2017-wang-et-al.pdf
65 SparseMatrix<Scalar> N, Z, M;
66 normal_derivative(V,T,N);
67 {
68 std::vector<Triplet<Scalar>> ZIJV;
69 for(int t =0;t<T.rows();t++)
70 {
71 for(int f =0;f<T.cols();f++)
72 {
73 if(C(t,f))
74 {
75 const int i = t+f*T.rows();
76 for(int c = 1;c<T.cols();c++)
77 {
78 ZIJV.emplace_back(T(t,(f+c)%T.cols()),i,1);
79 }
80 }
81 }
82 }
83 Z.resize(V.rows(),N.rows());
84 Z.setFromTriplets(ZIJV.begin(),ZIJV.end());
85 N = (Z*N).eval();
86 }
87 cotmatrix(V,T,L);
88 K = N+L;
89 massmatrix(V,T,MASSMATRIX_TYPE_DEFAULT,M);
90 // normalize
91 M /= ((Matrix<Scalar, Dynamic, 1>)M.diagonal()).array().abs().maxCoeff();
92 Minv =
93 ((Matrix<Scalar, Dynamic, 1>)M.diagonal().array().inverse()).asDiagonal();
94#else
95 Eigen::SparseMatrix<Scalar> M;
96 Eigen::Matrix<Integer, Dynamic, Dynamic> E;
97 Eigen::Matrix<Integer, Dynamic, 1> EMAP;
98 crouzeix_raviart_massmatrix(V,T,M,E,EMAP);
99 crouzeix_raviart_cotmatrix(V,T,E,EMAP,L);
100 // Ad #E by #V facet-vertex incidence matrix
101 Eigen::SparseMatrix<Scalar> Ad(E.rows(),V.rows());
102 {
103 std::vector<Eigen::Triplet<Scalar>> AIJV(E.size());
104 for(int e = 0;e<E.rows();e++)
105 {
106 for(int c = 0;c<E.cols();c++)
107 {
108 AIJV[e + c * E.rows()] = Eigen::Triplet<Scalar>(e, E(e, c), 1);
109 }
110 }
111 Ad.setFromTriplets(AIJV.begin(),AIJV.end());
112 }
113 // Degrees
114 Eigen::Matrix<Scalar, Dynamic, 1> De;
115 sum(Ad,2,De);
116 Eigen::DiagonalMatrix<Scalar,Eigen::Dynamic> De_diag =
117 De.array().inverse().matrix().asDiagonal();
118 K = L*(De_diag*Ad);
119 // normalize
120 M /= ((Matrix<Scalar, Dynamic, 1>)M.diagonal()).array().abs().maxCoeff();
121 Minv = ((Matrix<Scalar, Dynamic, 1>)M.diagonal().array().inverse()).asDiagonal();
122 // kill boundary edges
123 for(int f = 0;f<T.rows();f++)
124 {
125 for(int c = 0;c<T.cols();c++)
126 {
127 if(C(f,c))
128 {
129 const int e = EMAP(f+T.rows()*c);
130 Minv.diagonal()(e) = 0;
131 }
132 }
133 }
134
135#endif
136 switch(k)
137 {
138 default:
139 assert(false && "unsupported");
140 case 2:
141 // For C1 smoothness in 2D, one should use bi-harmonic
142 A = K.transpose() * (Minv * K);
143 break;
144 case 3:
145 // For C1 smoothness in 3D, one should use tri-harmonic
146 A = K.transpose() * (Minv * (-L * (Minv * K)));
147 break;
148 }
149 }
150 // Vertices in point handles
151 const size_t mp =
152 count_if(S.begin(),S.end(),[](const vector<int> & h){return h.size()==1;});
153 // number of region handles
154 const size_t r = S.size()-mp;
155 // Vertices in region handles
156 size_t mr = 0;
157 for(const auto & h : S)
158 {
159 if(h.size() > 1)
160 {
161 mr += h.size();
162 }
163 }
164 const size_t dim = T.cols()-1;
165 // Might as well be dense... I think...
166 Matrix<Scalar, Dynamic, Dynamic> J = Matrix<Scalar, Dynamic, Dynamic>::Zero(mp+mr,mp+r*(dim+1));
167 Matrix<Integer, Dynamic, 1> b(mp+mr);
168 Matrix<Scalar, Dynamic, Dynamic> H(mp+r*(dim+1),dim);
169 {
170 int v = 0;
171 int c = 0;
172 for(int h = 0;h<S.size();h++)
173 {
174 if(S[h].size()==1)
175 {
176 H.row(c) = V.block(S[h][0],0,1,dim);
177 J(v,c++) = 1;
178 b(v) = S[h][0];
179 v++;
180 }else
181 {
182 assert(S[h].size() >= dim+1);
183 for(int p = 0;p<S[h].size();p++)
184 {
185 for(int d = 0;d<dim;d++)
186 {
187 J(v,c+d) = V(S[h][p],d);
188 }
189 J(v,c+dim) = 1;
190 b(v) = S[h][p];
191 v++;
192 }
193 H.block(c,0,dim+1,dim).setIdentity();
194 c+=dim+1;
195 }
196 }
197 }
198 // minimize Â½ W' A W'
199 // subject to W(b,:) = J
200 return min_quad_with_fixed(
201 A,Matrix<Scalar, Dynamic, 1>::Zero(A.rows()).eval(),b,J,SparseMatrix<Scalar>(),Matrix<Scalar, Dynamic, 1>(),true,W);
202}
