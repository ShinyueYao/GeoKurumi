18template <typename Derivedl, typename DerivedF, typename Scalar>
19IGL_INLINE void igl::massmatrix_intrinsic(
20 const Eigen::MatrixBase<Derivedl> & l,
21 const Eigen::MatrixBase<DerivedF> & F,
22 const MassMatrixType type,
23 Eigen::SparseMatrix<Scalar>& M)
24{
25 const int n = F.maxCoeff()+1;
26 return massmatrix_intrinsic(l,F,type,n,M);
27}

29template <typename Derivedl, typename DerivedF, typename Scalar>
30IGL_INLINE void igl::massmatrix_intrinsic(
31 const Eigen::MatrixBase<Derivedl> & l,
32 const Eigen::MatrixBase<DerivedF> & F,
33 const MassMatrixType type,
34 const int n,
35 Eigen::SparseMatrix<Scalar>& M)
36{
37 using namespace Eigen;
38 using namespace std;
39 MassMatrixType eff_type = type;
40 const int m = F.rows();
41 const int simplex_size = F.cols();
42 // Use voronoi of for triangles by default, otherwise barycentric
43 if(type == MASSMATRIX_TYPE_DEFAULT)
44 {
45 eff_type = (simplex_size == 3?MASSMATRIX_TYPE_VORONOI:MASSMATRIX_TYPE_BARYCENTRIC);
46 }
47 assert(F.cols() == 3 && "only triangles supported");
48 Matrix<Scalar,Dynamic,1> dblA;
49 doublearea(l,0.,dblA);
50 Matrix<typename DerivedF::Scalar,Dynamic,1> MI;
51 Matrix<typename DerivedF::Scalar,Dynamic,1> MJ;
52 Matrix<Scalar,Dynamic,1> MV;
53
54 switch(eff_type)
55 {
56 case MASSMATRIX_TYPE_BARYCENTRIC:
57 // diagonal entries for each face corner
58 MI.resize(m*3,1); MJ.resize(m*3,1); MV.resize(m*3,1);
59 MI.block(0*m,0,m,1) = F.col(0);
60 MI.block(1*m,0,m,1) = F.col(1);
61 MI.block(2*m,0,m,1) = F.col(2);
62 MJ = MI;
63 repmat(dblA,3,1,MV);
64 MV.array() /= 6.0;
65 break;
66 case MASSMATRIX_TYPE_VORONOI:
67 {
68 // diagonal entries for each face corner
69 // http://www.alecjacobson.com/weblog/?p=874
70 MI.resize(m*3,1); MJ.resize(m*3,1); MV.resize(m*3,1);
71 MI.block(0*m,0,m,1) = F.col(0);
72 MI.block(1*m,0,m,1) = F.col(1);
73 MI.block(2*m,0,m,1) = F.col(2);
74 MJ = MI;
75
76 // Holy shit this needs to be cleaned up and optimized
77 Matrix<Scalar,Dynamic,3> cosines(m,3);
78 cosines.col(0) =
79 (l.col(2).array().pow(2)+l.col(1).array().pow(2)-l.col(0).array().pow(2))/(l.col(1).array()*l.col(2).array()*2.0);
80 cosines.col(1) =
81 (l.col(0).array().pow(2)+l.col(2).array().pow(2)-l.col(1).array().pow(2))/(l.col(2).array()*l.col(0).array()*2.0);
82 cosines.col(2) =
83 (l.col(1).array().pow(2)+l.col(0).array().pow(2)-l.col(2).array().pow(2))/(l.col(0).array()*l.col(1).array()*2.0);
84 Matrix<Scalar,Dynamic,3> barycentric = cosines.array() * l.array();
85 normalize_row_sums(barycentric,barycentric);
86 Matrix<Scalar,Dynamic,3> partial = barycentric;
87 partial.col(0).array() *= dblA.array() * 0.5;
88 partial.col(1).array() *= dblA.array() * 0.5;
89 partial.col(2).array() *= dblA.array() * 0.5;
90 Matrix<Scalar,Dynamic,3> quads(partial.rows(),partial.cols());
91 quads.col(0) = (partial.col(1)+partial.col(2))*0.5;
92 quads.col(1) = (partial.col(2)+partial.col(0))*0.5;
93 quads.col(2) = (partial.col(0)+partial.col(1))*0.5;
94
95 quads.col(0) = (cosines.col(0).array()<0).select( 0.25*dblA,quads.col(0));
96 quads.col(1) = (cosines.col(0).array()<0).select(0.125*dblA,quads.col(1));
97 quads.col(2) = (cosines.col(0).array()<0).select(0.125*dblA,quads.col(2));
98
99 quads.col(0) = (cosines.col(1).array()<0).select(0.125*dblA,quads.col(0));
100 quads.col(1) = (cosines.col(1).array()<0).select(0.25*dblA,quads.col(1));
101 quads.col(2) = (cosines.col(1).array()<0).select(0.125*dblA,quads.col(2));
102
103 quads.col(0) = (cosines.col(2).array()<0).select(0.125*dblA,quads.col(0));
104 quads.col(1) = (cosines.col(2).array()<0).select(0.125*dblA,quads.col(1));
105 quads.col(2) = (cosines.col(2).array()<0).select( 0.25*dblA,quads.col(2));
106
107 MV.block(0*m,0,m,1) = quads.col(0);
108 MV.block(1*m,0,m,1) = quads.col(1);
109 MV.block(2*m,0,m,1) = quads.col(2);
110
111 break;
112 }
113 case MASSMATRIX_TYPE_FULL:
114 assert(false && "Implementation incomplete");
115 break;
116 default:
117 assert(false && "Unknown Mass matrix eff_type");
118 }
119 sparse(MI,MJ,MV,n,n,M);
120}
