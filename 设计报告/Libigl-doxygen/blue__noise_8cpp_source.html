29 // Should probably find and replace with less generic name
30 //
31 // Map 3D subscripts (x,y,z) to unique index (return value)
32 //
33 // Inputs:
34 // w side length of w×w×w integer cube lattice
35 // x subscript along x direction
36 // y subscript along y direction
37 // z subscript along z direction
38 // Returns index value
39 //
40 inline BlueNoiseKeyType blue_noise_key(
41 const BlueNoiseKeyType w, // pass by copy --> int64_t so that multiplication is OK
42 const BlueNoiseKeyType x, // pass by copy --> int64_t so that multiplication is OK
43 const BlueNoiseKeyType y, // pass by copy --> int64_t so that multiplication is OK
44 const BlueNoiseKeyType z) // pass by copy --> int64_t so that multiplication is OK
45 {
46 return x+w*(y+w*z);
47 }
48 // Determine if a query candidate at position X.row(i) is too close to already
49 // selected sites (stored in S).
50 //
51 // Inputs:
52 // X #X by 3 list of raw candidate positions
53 // Xs #Xs by 3 list of corresponding integer cell subscripts
54 // i index of candidate in question
55 // S map from cell index to index into X of selected candidate (or -1 if
56 // cell is currently empty)
57 // rr Poisson disk radius squared
58 // w side length of w×w×w integer cube lattice (into which Xs subscripts)
59 template <
60 typename DerivedX,
61 typename DerivedXs>
62 inline bool blue_noise_far_enough(
63 const Eigen::MatrixBase<DerivedX> & X,
64 const Eigen::MatrixBase<DerivedXs> & Xs,
65 const std::unordered_map<BlueNoiseKeyType,int> & S,
66 const double & rr,
67 const int & w,
68 const int i)
69 {
70 const int xi = Xs(i,0);
71 const int yi = Xs(i,1);
72 const int zi = Xs(i,2);
73 BlueNoiseKeyType k = blue_noise_key(w,xi,yi,zi);
74 int g = 2; // ceil(r/s)
75 for(int x = std::max(xi-g,0);x<=std::min(xi+g,w-1);x++)
76 for(int y = std::max(yi-g,0);y<=std::min(yi+g,w-1);y++)
77 for(int z = std::max(zi-g,0);z<=std::min(zi+g,w-1);z++)
78 {
79 if(x!=xi || y!=yi || z!=zi)
80 {
81 const BlueNoiseKeyType nk = blue_noise_key(w,x,y,z);
82 // have already selected from this cell
83 const auto Siter = S.find(nk);
84 if(Siter !=S.end() && Siter->second >= 0)
85 {
86 const int ni = Siter->second;
87 // too close
88 if( (X.row(i)-X.row(ni)).squaredNorm() < rr)
89 {
90 return false;
91 }
92 }
93 }
94 }
95 return true;
96 }
97 // Try to activate a candidate in a given cell
98 //
99 // Inputs:
100 // X #X by 3 list of raw candidate positions
101 // Xs #Xs by 3 list of corresponding integer cell subscripts
102 // rr Poisson disk radius squared
103 // w side length of w×w×w integer cube lattice (into which Xs subscripts)
104 // nk index of cell in which we'd like to activate a candidate
105 // M map from cell index to list of candidates
106 // S map from cell index to index into X of selected candidate (or -1 if
107 // cell is currently empty)
108 // active list of indices into X of active candidates
109 // Outputs:
110 // M visited candidates deemed too close to already selected points are
111 // removed
112 // S updated to reflect activated point (if successful)
113 // active updated to reflect activated point (if successful)
114 // Returns true iff activation was successful
115 template <
116 typename DerivedX,
117 typename DerivedXs>
118 inline bool activate(
119 const Eigen::MatrixBase<DerivedX> & X,
120 const Eigen::MatrixBase<DerivedXs> & Xs,
121 const double & rr,
122 const int & i,
123 const int & w,
124 const BlueNoiseKeyType & nk,
125 std::unordered_map<BlueNoiseKeyType,std::vector<int> > & M,
126 std::unordered_map<BlueNoiseKeyType,int> & S,
127 std::vector<int> & active)
128 {
129 assert(M.count(nk));
130 auto & Mvec = M.find(nk)->second;
131 auto miter = Mvec.begin();
132 while(miter != Mvec.end())
133 {
134 const int mi = *miter;
135 // mi is our candidate sample. Is it far enough from all existing
136 // samples?
137 if(i>=0 && (X.row(i)-X.row(mi)).squaredNorm() > 4.*rr)
138 {
139 // too far skip (reject)
140 miter++;
141 } else if(blue_noise_far_enough(X,Xs,S,rr,w,mi))
142 {
143 active.push_back(mi);
144 S.find(nk)->second = mi;
145 //printf(" found %d\n",mi);
146 return true;
147 }else
148 {
149 // remove forever (instead of incrementing we swap and eat from the
150 // back)
151 //std::swap(*miter,Mvec.back());
152 *miter = Mvec.back();
153 bool was_last = (std::next(miter) == Mvec.end());
154 Mvec.pop_back();
155 if (was_last) {
156 // popping from the vector can invalidate the iterator, if it was
157 // pointing to the last element that was popped. Alternatively,
158 // one could use indices directly...
159 miter = Mvec.end();
160 }
161 }
162 }
163 return false;
164 }

166 template <
167 typename DerivedX,
168 typename DerivedXs>
169 inline bool step(
170 const Eigen::MatrixBase<DerivedX> & X,
171 const Eigen::MatrixBase<DerivedXs> & Xs,
172 const double & rr,
173 const int & w,
174 std::unordered_map<BlueNoiseKeyType,std::vector<int> > & M,
175 std::unordered_map<BlueNoiseKeyType,int> & S,
176 std::vector<int> & active,
177 std::vector<int> & collected
178 )
179 {
180 //considered.clear();
181 if(active.size() == 0) return false;
182 // random entry
183 const int e = rand() % active.size();
184 const int i = active[e];
185 //printf("%d\n",i);
186 const int xi = Xs(i,0);
187 const int yi = Xs(i,1);
188 const int zi = Xs(i,2);
189 //printf("%d %d %d - %g %g %g\n",xi,yi,zi,X(i,0),X(i,1),X(i,2));
190 // cell indices of neighbors
191 int g = 4;
192 std::vector<BlueNoiseKeyType> N;N.reserve((1+g*1)^3-1);
193 for(int x = std::max(xi-g,0);x<=std::min(xi+g,w-1);x++)
194 for(int y = std::max(yi-g,0);y<=std::min(yi+g,w-1);y++)
195 for(int z = std::max(zi-g,0);z<=std::min(zi+g,w-1);z++)
196 {
197 if(x!=xi || y!=yi || z!=zi)
198 {
199 //printf(" %d %d %d\n",x,y,z);
200 const BlueNoiseKeyType nk = blue_noise_key(w,x,y,z);
201 // haven't yet selected from this cell?
202 const auto Siter = S.find(nk);
203 if(Siter !=S.end() && Siter->second < 0)
204 {
205 assert(M.find(nk) != M.end());
206 N.emplace_back(nk);
207 }
208 }
209 }
210 //printf(" --------\n");
211 // randomize order: this might be a little paranoid...
212 std::random_shuffle(std::begin(N), std::end(N));
213 bool found = false;
214 for(const BlueNoiseKeyType & nk : N)
215 {
216 assert(M.find(nk) != M.end());
217 if(activate(X,Xs,rr,i,w,nk,M,S,active))
218 {
219 found = true;
220 break;
221 }
222 }
223 if(!found)
224 {
225 // remove i from active list
226 // https://stackoverflow.com/a/60765833/148668
227 collected.push_back(i);
228 //printf(" before: "); for(const int j : active) { printf("%d ",j); } printf("\n");
229 std::swap(active[e], active.back());
230 //printf(" after : "); for(const int j : active) { printf("%d ",j); } printf("\n");
231 active.pop_back();
232 //printf(" removed %d\n",i);
233 }
234 //printf(" active: "); for(const int j : active) { printf("%d ",j); } printf("\n");
235 return true;
236 }
237}

239template <
240 typename DerivedV,
241 typename DerivedF,
242 typename DerivedB,
243 typename DerivedFI,
244 typename DerivedP>
245IGL_INLINE void igl::blue_noise(
246 const Eigen::MatrixBase<DerivedV> & V,
247 const Eigen::MatrixBase<DerivedF> & F,
248 const typename DerivedV::Scalar r,
249 Eigen::PlainObjectBase<DerivedB> & B,
250 Eigen::PlainObjectBase<DerivedFI> & FI,
251 Eigen::PlainObjectBase<DerivedP> & P)
252{
253 typedef typename DerivedV::Scalar Scalar;
254 typedef Eigen::Matrix<Scalar,Eigen::Dynamic,1> VectorXS;
255 // float+RowMajor is faster...
256 typedef Eigen::Matrix<Scalar,Eigen::Dynamic,3,Eigen::RowMajor> MatrixX3S;
257 assert(V.cols() == 3 && "Only 3D embeddings allowed");
258 // minimum radius
259 const Scalar min_r = r;
260 // cell size based on 3D distance
261 // It works reasonably well (but is probably biased to use s=2*r/√3 here and
262 // g=1 in the outer loop below.
263 //
264 // One thing to try would be to store a list in S (rather than a single point)
265 // or equivalently a mask over M and just use M as a generic spatial hash
266 // (with arbitrary size) and then tune its size (being careful to make g a
267 // function of r and s; and removing the `if S=-1 checks`)
268 const Scalar s = r/sqrt(3.0);
269
270 const double area =
271 [&](){Eigen::VectorXd A;igl::doublearea(V,F,A);return A.array().sum()/2;}();
272 // Circle packing in the plane has igl::PI*sqrt(3)/6 efficiency
273 const double expected_number_of_points =
274 area * (igl::PI * sqrt(3.0) / 6.0) / (igl::PI * min_r * min_r / 4.0);
275
276 // Make a uniform random sampling with 30*expected_number_of_points.
277 const int nx = 30.0*expected_number_of_points;
278 MatrixX3S X,XB;
279 Eigen::VectorXi XFI;
280 igl::random_points_on_mesh(nx,V,F,XB,XFI,X);
281
282 // Rescale so that s = 1
283 Eigen::Matrix<int,Eigen::Dynamic,3,Eigen::RowMajor> Xs =
284 ((X.rowwise()-X.colwise().minCoeff())/s).template cast<int>();
285 const int w = Xs.maxCoeff()+1;
286 {
287 Eigen::VectorXi I;
288 igl::sortrows(decltype(Xs)(Xs),true,Xs,I);
289 igl::slice(decltype(X)(X),I,1,X);
290 // These two could be spun off in their own thread.
291 igl::slice(decltype(XB)(XB),I,1,XB);
292 igl::slice(decltype(XFI)(XFI),I,1,XFI);
293 }
294 // Initialization
295 std::unordered_map<BlueNoiseKeyType,std::vector<int> > M;
296 std::unordered_map<BlueNoiseKeyType, int > S;
297 // attempted to seed
298 std::unordered_map<BlueNoiseKeyType, int > A;
299 // Q: Too many?
300 // A: Seems to help though.
301 M.reserve(Xs.rows());
302 S.reserve(Xs.rows());
303 for(int i = 0;i<Xs.rows();i++)
304 {
305 BlueNoiseKeyType k = blue_noise_key(w,Xs(i,0),Xs(i,1),Xs(i,2));
306 const auto Miter = M.find(k);
307 if(Miter == M.end())
308 {
309 M.insert({k,{i}});
310 }else
311 {
312 Miter->second.push_back(i);
313 }
314 S.emplace(k,-1);
315 A.emplace(k,false);
316 }
317
318 std::vector<int> active;
319 // precompute r²
320 // Q: is this necessary?
321 const double rr = r*r;
322 std::vector<int> collected;
323 collected.reserve(2.0*expected_number_of_points);
324
325 auto Mouter = M.begin();
326 // Just take the first point as the initial seed
327 const auto initialize = [&]()->bool
328 {
329 while(true)
330 {
331 if(Mouter == M.end())
332 {
333 return false;
334 }
335 const BlueNoiseKeyType k = Mouter->first;
336 // Haven't placed in this cell yet
337 if(S[k]<0)
338 {
339 if(activate(X,Xs,rr,-1,w,k,M,S,active)) return true;
340 }
341 Mouter++;
342 }
343 assert(false && "should not be reachable.");
344 };
345
346 // important if mesh contains many connected components
347 while(initialize())
348 {
349 while(active.size()>0)
350 {
351 step(X,Xs,rr,w,M,S,active,collected);
352 }
353 }
354 {
355 const int n = collected.size();
356 P.resize(n,3);
357 B.resize(n,3);
358 FI.resize(n);
359 for(int i = 0;i<n;i++)
360 {
361 const int c = collected[i];
362 P.row(i) = X.row(c).template cast<typename DerivedP::Scalar>();
363 B.row(i) = XB.row(c).template cast<typename DerivedB::Scalar>();
364 FI(i) = XFI(c);
365 }
366 }
367}
