11 const Eigen::MatrixBase<DerivedB> & B,
12 const Eigen::MatrixBase<DerivedP> & P)
13{
14 typedef typename DerivedA::Scalar SType;
15 // Gather vectors to source and destination
16 SType o2A[2];
17 SType o2B[2];
18 // and lengths
19 SType o2Al = 0;
20 SType o2Bl = 0;
21 for(int i = 0;i<2;i++)
22 {
23 o2A[i] = P(i) - A(i);
24 o2B[i] = P(i) - B(i);
25 o2Al += o2A[i]*o2A[i];
26 o2Bl += o2B[i]*o2B[i];
27 }
28 o2Al = sqrt(o2Al);
29 o2Bl = sqrt(o2Bl);
30 // Normalize
31 for(int i = 0;i<2;i++)
32 {
33 // Matlab crashes on NaN
34 if(o2Al!=0)
35 {
36 o2A[i] /= o2Al;
37 }
38 if(o2Bl!=0)
39 {
40 o2B[i] /= o2Bl;
41 }
42 }
43 return
44 -atan2(o2B[0]*o2A[1]-o2B[1]*o2A[0],o2B[0]*o2A[0]+o2B[1]*o2A[1])/
45 (2.*igl::PI);
46}
