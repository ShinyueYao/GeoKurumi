1612IGL_INLINE void igl::colormap(
1613 const double palette[256][3], const T x_in, T & r, T & g, T & b)
1614{
1615 static const unsigned int pal = 256;
1616 const T zero = 0.0;
1617 const T one = 1.0;
1618 T x_in_clamped = static_cast<T>(std::max(zero, std::min(one, x_in)));
1619
1620 // simple rgb lerp from palette
1621 unsigned int least = std::floor(x_in_clamped * static_cast<T>(pal - 1));
1622 unsigned int most = std::ceil(x_in_clamped * static_cast<T>(pal - 1));
1623
1624 T _r[2] = { static_cast<T>(palette[least][0]), static_cast<T>(palette[most][0]) };
1625 T _g[2] = { static_cast<T>(palette[least][1]), static_cast<T>(palette[most][1]) };
1626 T _b[2] = { static_cast<T>(palette[least][2]), static_cast<T>(palette[most][2]) };
1627
1628 T t = std::max(zero, std::min(one, static_cast<T>(fmod(x_in_clamped * static_cast<T>(pal), one))));
1629
1630 r = std::max(zero, std::min(one, (one - t) * _r[0] + t * _r[1]));
1631 g = std::max(zero, std::min(one, (one - t) * _g[0] + t * _g[1]));
1632 b = std::max(zero, std::min(one, (one - t) * _b[0] + t * _b[1]));
1633}
