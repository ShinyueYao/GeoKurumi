37template <
38 typename Derivedl_in,
39 typename DerivedF_in,
40 typename Derivedl,
41 typename DerivedF,
42 typename DerivedE,
43 typename DeriveduE,
44 typename DerivedEMAP,
45 typename uE2EType>
46IGL_INLINE void igl::intrinsic_delaunay_triangulation(
47 const Eigen::MatrixBase<Derivedl_in> & l_in,
48 const Eigen::MatrixBase<DerivedF_in> & F_in,
49 Eigen::PlainObjectBase<Derivedl> & l,
50 Eigen::PlainObjectBase<DerivedF> & F,
51 Eigen::PlainObjectBase<DerivedE> & E,
52 Eigen::PlainObjectBase<DeriveduE> & uE,
53 Eigen::PlainObjectBase<DerivedEMAP> & EMAP,
54 std::vector<std::vector<uE2EType> > & uE2E)
55{
56 igl::unique_edge_map(F_in, E, uE, EMAP, uE2E);
57 // We're going to work in place
58 l = l_in;
59 F = F_in;
60 typedef typename DerivedF::Scalar Index;
61 typedef typename Derivedl::Scalar Scalar;
62 const Index num_faces = F.rows();
63
64 // Vector is faster than queue...
65 std::vector<Index> Q;
66 Q.reserve(uE2E.size());
67 for (size_t uei=0; uei<uE2E.size(); uei++)
68 {
69 Q.push_back(uei);
70 }
71 // I tried using a "delaunay_since = iter" flag to avoid duplicates, but there
72 // was no speed up.
73
74 while(!Q.empty())
75 {
76#ifdef IGL_INTRINSIC_DELAUNAY_TRIANGULATION_DEBUG
77 // Expensive sanity check
78 {
79 Eigen::Matrix<bool,Eigen::Dynamic,1> inQ(uE2E.size(),1);
80 inQ.setConstant(false);
81 for(const auto uei : Q)
82 {
83 inQ(uei) = true;
84 }
85 for (Index uei=0; uei<uE2E.size(); uei++)
86 {
87 if(!inQ(uei) && !is_intrinsic_delaunay(l,uE2E,num_faces,uei))
88 {
89 std::cout<<" "<<uei<<" might never be fixed!"<<std::endl;
90 }
91 }
92 }
93#endif
94
95 const Index uei = Q.back();
96 Q.pop_back();
97 if (uE2E[uei].size() == 2)
98 {
99 if(!is_intrinsic_delaunay(l,uE2E,num_faces,uei))
100 {
101 // update l just before flipping edge
102 // . //
103 // /|\ //
104 // a/ | \d //
105 // / e \ //
106 // / | \ //
107 // .----|-f--. //
108 // \ | / //
109 // \ | / //
110 // b\α|δ/c //
111 // \|/ //
112 // . //
113 // Annotated from flip_edge:
114 // Edge to flip [v1,v2] --> [v3,v4]
115 // Before:
116 // F(f1,:) = [v1,v2,v4] // in some cyclic order
117 // F(f2,:) = [v1,v3,v2] // in some cyclic order
118 // After:
119 // F(f1,:) = [v1,v3,v4] // in *this* order
120 // F(f2,:) = [v2,v4,v3] // in *this* order
121 //
122 // v1 v1
123 // /|\ / \
124 // c/ | \b c/f1 \b
125 // v3 /f2|f1\ v4 => v3 /__f__\ v4
126 // \ e / \ f2 /
127 // d\ | /a d\ /a
128 // \|/ \ /
129 // v2 v2
130 //
131 // Compute intrinsic length of oppposite edge
132 assert(uE2E[uei].size() == 2 && "edge should have 2 incident faces");
133 const Index f1 = uE2E[uei][0]%num_faces;
134 const Index f2 = uE2E[uei][1]%num_faces;
135 const Index c1 = uE2E[uei][0]/num_faces;
136 const Index c2 = uE2E[uei][1]/num_faces;
137 assert(c1 < 3);
138 assert(c2 < 3);
139 assert(f1 != f2);
140 const Index v1 = F(f1, (c1+1)%3);
141 const Index v2 = F(f1, (c1+2)%3);
142 const Index v4 = F(f1, c1);
143 const Index v3 = F(f2, c2);
144 assert(F(f2, (c2+2)%3) == v1);
145 assert(F(f2, (c2+1)%3) == v2);
146 assert( std::abs(l(f1,c1)-l(f2,c2)) < igl::EPS<Scalar>() );
147 const Scalar e = l(f1,c1);
148 const Scalar a = l(f1,(c1+1)%3);
149 const Scalar b = l(f1,(c1+2)%3);
150 const Scalar c = l(f2,(c2+1)%3);
151 const Scalar d = l(f2,(c2+2)%3);
152 // tan(α/2)
153 const Scalar tan_a_2= tan_half_angle(a,b,e);
154 // tan(δ/2)
155 const Scalar tan_d_2 = tan_half_angle(d,e,c);
156 // tan((α+δ)/2)
157 const Scalar tan_a_d_2 = (tan_a_2 + tan_d_2)/(1.0-tan_a_2*tan_d_2);
158 // cos(α+δ)
159 const Scalar cos_a_d =
160 (1.0 - tan_a_d_2*tan_a_d_2)/(1.0+tan_a_d_2*tan_a_d_2);
161 const Scalar f = sqrt(b*b + c*c - 2.0*b*c*cos_a_d);
162 l(f1,0) = f;
163 l(f1,1) = b;
164 l(f1,2) = c;
165 l(f2,0) = f;
166 l(f2,1) = d;
167 l(f2,2) = a;
168 // Important to grab these indices _before_ calling flip_edges (they
169 // will be correct after)
170 const size_t e_24 = f1 + ((c1 + 1) % 3) * num_faces;
171 const size_t e_41 = f1 + ((c1 + 2) % 3) * num_faces;
172 const size_t e_13 = f2 + ((c2 + 1) % 3) * num_faces;
173 const size_t e_32 = f2 + ((c2 + 2) % 3) * num_faces;
174 const size_t ue_24 = EMAP(e_24);
175 const size_t ue_41 = EMAP(e_41);
176 const size_t ue_13 = EMAP(e_13);
177 const size_t ue_32 = EMAP(e_32);
178 flip_edge(F, E, uE, EMAP, uE2E, uei);
179 Q.push_back(ue_24);
180 Q.push_back(ue_41);
181 Q.push_back(ue_13);
182 Q.push_back(ue_32);
183 }
184 }
185 }
186}
