26template<
27 typename DerivedQ,
28 typename DerivedE>
29IGL_INLINE void igl::quad_grid(
30 const int nx,
31 const int ny,
32 Eigen::PlainObjectBase<DerivedQ> & Q,
33 Eigen::PlainObjectBase<DerivedE> & E)
34{
35 Eigen::MatrixXi I(nx,ny);
36 Q.resize( (nx-1)*(ny-1),4);
37 E.resize((nx-1)*ny + (ny-1)*nx,2);
38 {
39 int v = 0;
40 int q = 0;
41 int e = 0;
42 // Ordered to match igl::grid
43 for(int y = 0;y<ny;y++)
44 {
45 for(int x = 0;x<nx;x++)
46 {
48 //V(v,0) = (-1.0) + double(x)/double(nx-1) * (1.0 - (-1.0));
49 //V(v,2) = (-1.0) + double(y)/double(ny-1) * (1.0 - (-1.0));
50 I(x,y) = v;
51 v++;
52 // Add a verical edge
53 if(y>0)
54 {
55 E(e,0) = I(x,y);
56 E(e,1) = I(x,y-1);
57 e++;
58 }
59 // Add a horizontal edge
60 if(x>0)
61 {
62 E(e,0) = I(x,y);
63 E(e,1) = I(x-1,y);
64 e++;
65 }
66 // Add two triangles
67 if(x>0 && y>0)
68 {
69 // -1,0----0,0
70 // | / |
71 // | / |
72 // | / |
73 // | / |
74 // -1,-1---0,-1
75 Q(q,0) = I(x-0,y-0);
76 Q(q,1) = I(x-1,y-0);
77 Q(q,2) = I(x-1,y-1);
78 Q(q,3) = I(x-0,y-1);
79 q++;
80 //F(f,2) = I(x-0,y-0);
81 //F(f,1) = I(x-1,y-0);
82 //F(f,0) = I(x-1,y-1);
83 //f++;
84 //F(f,2) = I(x-0,y-0);
85 //F(f,1) = I(x-1,y-1);
86 //F(f,0) = I(x-0,y-1);
87 //f++;
88 }
89 }
90 }
91 }
92}
