6template <typename Scalar, int n, int m>
7IGL_INLINE Eigen::Matrix<Scalar,n,1> igl::quadprog(
8 const Eigen::Matrix<Scalar,n,n> & H,
9 const Eigen::Matrix<Scalar,n,1> & f,
10 const Eigen::Matrix<Scalar,m,n> & A,
11 const Eigen::Matrix<Scalar,m,1> & b,
12 const Eigen::Matrix<Scalar,n,1> & lb,
13 const Eigen::Matrix<Scalar,n,1> & ub)
14{
15 // Alec 16/2/2021:
16 // igl::quadprog implements a very simple primal active set method. The new
17 // igl::min_quad_with_fixed is very fast for small dense problems so the
18 // iterations of igl::quadprog become very fast. Even if it ends up doing many
19 // more iterations than igl::copyleft::quadprog it would be much faster (in
20 // reality it doesn't do that many more iterations). It's a healthy 10-100x
21 // faster than igl::copyleft::quadprog for specific cases of QPs.
22 //
23 // Unfortunately, that set is limited. igl::quadprog is really good at tiny
24 // box-constrained QPs with a positive definite objective (like the kind that show
25 // up in dual contouring). igl::copyleft::quadprog handles more general problems
26 // (and also starts to beat igl::quadprog when the number of variables gets over
27 // ~20). I tried extending igl::quadprog so that we could use it for
28 // igl::copyleft::progressive_hulls and drop igl::copyleft::quadprog but it was
29 // trickier than I thought. Something like qpmad or the non GPL version of
30 // quadrog++ would be good future PR.
31 //
32 typedef Eigen::Matrix<Scalar,n,1> VectorSn;
33 typedef Eigen::Array<bool,n,1> Arraybn;
34 assert( (lb.array() < ub.array() ).all() );
35 const int dyn_n = n == Eigen::Dynamic ? H.rows() : n;
36 VectorSn x(dyn_n);
37 VectorSn bc = VectorSn::Constant(dyn_n,1,-1e26);
38 Arraybn k = Arraybn::Constant(dyn_n,1,false);
39 Eigen::Index iter;
40 // nÂ³ is probably way too conservative.
41 for(iter = 0;iter<dyn_n*dyn_n*dyn_n;iter++)
42 {
43 // For dual contouring 99% of the time the active set is empty.
44 // Optimize for this common case.
45 // Windows needs template arguments spelled out
46 x = min_quad_with_fixed<Scalar,n,m>(H,f,k,bc,A,b);
47 // constraint violations
48 VectorSn vl = lb-x;
49 VectorSn vu = x-ub;
50
51 // try to add/remove constraints
52 Eigen::Index best_add = -1; Scalar worst_offense = 0;
53 bool add_lower;
54 Eigen::Index best_remove = -1; Scalar worst_lambda = 0;
55 for(Eigen::Index i = 0;i<dyn_n;i++)
56 {
57 if(vl(i)>worst_offense)
58 {
59 best_add = i;
60 add_lower = true;
61 worst_offense = vl(i);
62 }
63 if(vu(i)>worst_offense)
64 {
65 best_add = i;
66 add_lower = false;
67 worst_offense = vu(i);
68 }
69 // bias toward adding constraints
70 if(best_add<0 && k(i))
71 {
72 const Scalar sign = bc(i)==ub(i)?1:-1;
73 const Scalar lambda_i = sign * (H.row(i)*x+f(i));
74 if(lambda_i > worst_lambda)
75 {
76 best_remove = i;
77 worst_lambda = lambda_i;
78 }
79 }
80 }
81 // bias toward adding constraints
82 if(best_add >= 0)
83 {
84 const auto i = best_add;
85 assert(!k(i));
86 bc(i) = add_lower ? lb(i) : ub(i);
87 k(i) = true;
88 }else if(best_remove >= 0)
89 {
90 const auto i = best_remove;
91 assert(k(i));
92 k(i) = false;
93 }else /*if(best_add < 0 && best_remove < 0)*/
94 {
95 return x;
96 }
97 }
98 // Should never happen.
99 assert(false && "quadprog failed after too many iterations");
100 return VectorSn::Zero(dyn_n);
101}
